27      public void tayta() {
28          sisalto.setLayout(new BorderLayout());
29  
30          JPanel content = new JPanel();
31  
32          JLabel teksti = new JLabel();
33          teksti.setText("<html><body align=\"center\"><strong>Tervetuloa pelaamaan Sudokua!</strong><br><br>"
34                  + "Valitse pelin vaikeusaste ja klikkaa Aloita peli aloittaaksesi uuden Sudoku-pelin.<br>"
35                  + "Jos haluat lisää haastetta, niin valitse Käytä aikarajaa,<br>"
36                  + "jolloin sinun täytyy ratkaista Sudoku annetussa määräajassa.<br>"
37                  + "Pelilaudan ruutujen värikoodaus on kuvattu alla näkyvässä taulukossa.</body></html>");
38          content.add(teksti);
39  
40          lisaaPainikkeet(content);
41          JCheckBox valinta = new JCheckBox("Käytä aikarajaa");
42          valinta.addItemListener(new Aikarajakasittelija(kali));
43          content.add(valinta);
44          
45          JLabel varit = new JLabel();
46          varit.setText("<html>\n"
47                  + "<head>\n"
48                  + "<style>\n"
49                  + "table, th, td {\n"

52                  + "}\n"
53                  + "</style>\n"
54                  + "</head>\n"
55                  + "<body>\n"
56                  + "<table style=\"border:solid;text-align:left\">\n"
57                  + "  <tr>\n"
58                  + "    <th>Väri</th>\n"
59                  + "    <th>Selite</th>\n"
60                  + "  </tr>\n"
61                  + "  <tr>\n"
62                  + "    <td style=\"background-color:white\"></td>\n"

81          content.add(varit);
82  
83          JPanel alapalkki = new JPanel();
84          JButton uusiPeli = new JButton("Aloita peli");
85          UusiPeliKasittelija kasittelija = new UusiPeliKasittelija(hallinta, kali);
86          uusiPeli.addActionListener(kasittelija);
87          alapalkki.add(uusiPeli);
88  
89          JPanel tyhjaa = new JPanel();
90          tyhjaa.setPreferredSize(new Dimension(600, 100));
91  
92          sisalto.add(tyhjaa, BorderLayout.NORTH);
93          sisalto.add(content, BorderLayout.CENTER);
94          sisalto.add(alapalkki, BorderLayout.SOUTH);
95      }

97      private void lisaaPainikkeet(Container content) {
98          JLabel otsikko = new JLabel("Vaikeusaste:");
99          ButtonGroup painikkeet = new ButtonGroup();
100 
101         JRadioButton helppo = new JRadioButton("Helppo (3 min)");
102         helppo.setActionCommand("1");
103 
104         JRadioButton keskitaso = new JRadioButton("Keskitaso (10 min)");
105         keskitaso.setActionCommand("2");
106 
107         JRadioButton vaikea = new JRadioButton("Vaikea (25 min)");
108         vaikea.setActionCommand("3");
109 
110         VaikeusasteKasittelija v = new VaikeusasteKasittelija(kali);
111         helppo.addActionListener(v);
112         keskitaso.addActionListener(v);
113         vaikea.addActionListener(v);
114 
115         painikkeet.add(helppo);
116         painikkeet.add(keskitaso);
117         painikkeet.add(vaikea);
118 
119         JPanel painikkeetContainer = new JPanel();
120         painikkeetContainer.add(otsikko);
121         painikkeetContainer.add(helppo);
122         painikkeetContainer.add(keskitaso);
123         painikkeetContainer.add(vaikea);
124         content.add(painikkeetContainer);
125     }
