4   import javax.swing.JOptionPane;
5   
6   
7   public class Ilmoitus {
8       private String[] painikkeet;
9       private String teksti;
10      private String otsikko;
11      private Nakymanhallinta hallinta;
12  
13      public Ilmoitus(String[] painikkeet, String teksti, String otsikko, Nakymanhallinta hallinta) {
14          this.painikkeet = painikkeet;
15          this.teksti = teksti;
16          this.otsikko = otsikko;
17          this.hallinta = hallinta;
18      }
19      
20      public void nayta() {
21          int vastaus = JOptionPane.showOptionDialog(hallinta.getFrame(), teksti, otsikko, JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null, painikkeet,
22                  painikkeet[0]);
23          kasitteleVastaus(vastaus);
24      }
25      
26      private void kasitteleVastaus(int vastaus) {
27          if (vastaus == 1) {
28              hallinta.naytaAloitusnaytto();
29          } else {
30              System.exit(0);
31          }
32      }
33  }
