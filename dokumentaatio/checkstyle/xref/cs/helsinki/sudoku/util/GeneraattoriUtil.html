11      public static ArrayList<AbstractMap.SimpleEntry<Integer, Integer>> annaListaRuutuja() {
12          ArrayList<Integer> rivit = yhdeksanLukua();
13          ArrayList<Integer> sarakkeet = yhdeksanLukua();
14          ArrayList<AbstractMap.SimpleEntry<Integer, Integer>> tulos = new ArrayList<>();
15          for (Integer rivi : rivit) {
16              for (Integer sarake : sarakkeet) {
17                  AbstractMap.SimpleEntry<Integer, Integer> lukupari = new AbstractMap.SimpleEntry<>(rivi - 1, sarake - 1);
18                  tulos.add(lukupari);
19              }
20          }
21          Collections.shuffle(tulos);
22          return tulos;
23      }

25      public static int[][] kloonaa(int[][] alkup) {
26          if (alkup == null) {
27              return null;
28          }
29  
30          final int[][] tulos = new int[alkup.length][];
31          for (int i = 0; i < alkup.length; i++) {
32              tulos[i] = Arrays.copyOf(alkup[i], alkup[i].length);
33          }
34          return tulos;
35      }

37      public static int haeOsaruutu(int rivi, int sarake) {
38          int osaruudunRivi = haeOsaruudunAlku(rivi);
39          int osaruudunSarake = haeOsaruudunAlku(sarake);
40          return osaruudunRivi + osaruudunSarake / 3;
41      }
42  
43      public static int haeOsaruudunAlku(int riviTaiSarake) {
44          return (riviTaiSarake / 3) * 3;
45      }
46  
47      public static boolean onRatkaistu(int[][] lauta) {
48          for (int i = 0; i < 9; i++) {
49              for (int j = 0; j < 9; j++) {
50                  if (lauta[i][j] == 0) {
51                      return false;
52                  }
53              }
54          }
55          return true;
56      }
57  
58      public static boolean tyhjaRuutu(int[][] lauta, int rivi, int sarake) {
59          return lauta[rivi][sarake] == 0;
60      }
61  
62      public static int tyhjienLkm(int[][] lauta) {
63          int lkm = 0;
64          for (int i = 0; i < 9; i++) {
65              for (int j = 0; j < 9; j++) {
66                  if (tyhjaRuutu(lauta, i, j)) {
67                      lkm++;
68                  }
69              }
70          }
71          return lkm;
72      }
